git status - (проверка статуса репозитория!)

git clone {ссылка} - (клонирование репозитория!)

cd {название папки}

git add {название файла} - (добавление одного файла)
git add . - (добавление всех файлов)

git commit -am "feat: make test repository" - (добавление)
git commit -am "fix: make test repository" - (изменение чего то)

git push - (отправка изменений)

git branch - (посмотреть ветки)

git checkout -b {название ветки} - (создание и переключение на ветку)

git branch {название ветки} - (просто создать ветку)

git checkout {название ветки} - (переключение на ветку)

git merge {название ветки с которой происходит слияние} - (слияние веток)

git push --set-upstream origin {название ветки} - (отправка локальной ветки репозитория в глобальную)

git branch -d {название ветки} - (удаление ветки)
